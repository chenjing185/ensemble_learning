            function myFun (a,b) {
                //升序
                return a-b;//根据返回值判断是否交换位置,40-5>0要换,<=0不换
                //降序
                //return b-a;
            }
            // 创建数组: var  数组名称 = new Array(size);
            var hobby=new Array();
            hobby[0]="10";
            hobby[1]="99";
            // push()   向数组的末尾添加一个或更多元素，并返回新的长度。
            hobby.push("3");
            console.log(hobby);
            for (var i = 0; i < hobby.length; i++) {
                console.log(hobby[i]);//10 99 3
            }
            // join()   把数组的所有元素放入一个字符串。元素通过指定的分隔符进行分隔。
            var joinStr=hobby.join(" ");
            //输出
            console.log(joinStr);
            
            // sort()   对数组的元素进行排序
            hobby.sort(myFun);//升序,从小到大
            console.log(hobby);
            //如果想要让数组变成从大到小.可以对数组颠倒
            // reverse()    颠倒数组中元素的顺序
            hobby.reverse();
            console.log(hobby);
            
            
            //数组的遍历
            console.log("遍历方式1:");
            for (var i = 0; i < hobby.length; i++) {
                console.log(hobby[i]);
            }
            console.log("遍历方式2:");
            for(index in hobby){
                console.log(index+":"+hobby[index]);
            }
            console.log("遍历方式3:");
            hobby.forEach(function  (value,index) {
                console.log(index+":"+value);
            })
        </script>

小案例:实现全选反选全不选

 
x
<body>
        <input type="checkbox" name="hobby"  value="" />吃
        <input type="checkbox" name="hobby"  value="" />喝
        <input type="checkbox" name="hobby"  value="" />玩
        <input type="checkbox" name="hobby"  value="" />睡
        <br/>
        <input type="button" id="checkAll" value="全选" />
        <input type="button" id="checkRe" value="反选" />
        <input type="button" id="checkNo" value="全不选" />
    </body>
    <script type="text/javascript">
        var checkAll=document.getElementById("checkAll");
        var checkRe=document.getElementById("checkRe");
        var checkNo=document.getElementById("checkNo");
        //全选
        checkAll.onclick=function  () {
            //获取复选框的对象,获取到的是一个数组对象
            var arr=document.getElementsByName("hobby");
            for (var i = 0; i < arr.length; i++) {
                  arr[i].checked = true;
                }
        }
        
        //全不选
        checkNo.onclick=function  () {
            //获取复选框的对象,获取到的是一个数组对象
            var arr=document.getElementsByName("hobby");
            for (var i = 0; i < arr.length; i++) {
                  arr[i].checked = false;
                }
        }
        
        //反选 把true改成false  把false改成true
        checkRe.onclick=function  () {
            //获取复选框的对象,获取到的是一个数组对象
            var arr=document.getElementsByName("hobby");
            for (var i = 0; i < arr.length; i++) {
                   arr[i].checked =!(arr[i].checked);
                }
        }
    </script>

2.Date对象

Date 对象用于处理日期和时间

可以通过new关键词来定义Date

对象的创建:var 对象名=new Date();

setTimeout("函数名()",间隔时间) 在指定的毫秒数后调用函数或计算表达式 调用一次
setInterval("函数名()",间隔时间)按照指定的周期（以毫秒计）来调用函数或表达式 调用多次

3.倒计时跳转页面

 
xxxxxxxxxx
<script type="text/javascript">
            var time=5;
            function show () {
                document.getElementById("time").innerHTML=time;
                time--;
                if(time==0){
                    //跳转页面
                    location.href="http://www.baidu.com";
                }else{
                    //定时器
                    setTimeout("show ()",1000);
                }
                
            }
        </script>
    </head>
    <body onload="show ()">
        <span id="time"></span>秒后跳转到百度首页
    </body>

3.JavaScript HTML DOM

通过 HTML DOM，可访问 JavaScript HTML 文档的所有元素。

HTML DOM (文档对象模型)

当网页被加载时，浏览器会创建页面的文档对象模型（Document Object Model）。

HTML DOM 模型被构造为对象的树：

HTML DOM 树

通过可编程的对象模型，JavaScript 获得了足够的能力来创建动态的 HTML。

  • JavaScript 能够改变页面中的所有 HTML 元素
  • JavaScript 能够改变页面中的所有 HTML 属性
  • JavaScript 能够改变页面中的所有 CSS 样式
  • JavaScript 能够对页面中的所有事件做出反应

1.查找 HTML 元素

通常，通过 JavaScript，您需要操作 HTML 元素。

为了做到这件事情，您必须首先找到该元素。有三种方法来做这件事：

  • 通过 id 找到 HTML 元素
  • 通过标签名找到 HTML 元素
  • 通过类名找到 HTML 元素
  • 通过name属性找到HTML元素
