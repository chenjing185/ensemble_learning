4.尝试使用jq实现省市级联

 
xxxxxxxxxx
<script type="text/javascript">
            $(function  () {
                    //1.将基本城市信息存放到数组中
                    var provinces=new Array("福建省","广东省","广西省");
                    provinces["福建省"]=["厦门市","漳州市","泉州市","龙岩市"];
                    provinces["广东省"]=["广州市","东莞市","揭阳市","潮汕市"];
                    provinces["广西省"]=["桂林市","北海市","柳州市"];
                    
                    var country=new Array();
                    country["厦门市"]=["思明区","湖里区","集美区"];
                    country["漳州市"]=["龙文区","芗城区","龙海市"];
                    country["广州市"]=["白云区","越秀区"];
                    
                    //2.将provinces数组添加到省份列表框
                    $.each(provinces, function(i,v) {
                        $("#province").append("<option value="+v+">"+v+"</option>");
                    });
                    
                    //3.province省份选项改变时的事件
                    $("#province").change(function  () {
                        //先将city清空,保留默认的default
                        $("#city")[0].length=1;
                        $("#country")[0].length=1;
                        
                        //this表示的是province下拉框
                        //this.value得到下拉框的值
                        $.each(provinces[this.value], function(i,v) {
                            $("#city").append("<option value="+v+">"+v+"</option>");
                        });
                        
                    });
                    //4.city城市选项改变时的事件
                    $("#city").change(function  () {
                        //先将city清空,保留默认的default
                        $("#country")[0].length=1;
                        
                        $.each(country[this.value], function(i,v) {
                            $("#country").append("<option value="+v+">"+v+"</option>");
                        });
                        
                    }); 
                    
            })
                
        </script>

7.思考题:jq操作表格增加行删除行

 
x
        $(function  () {
            //添加订单
            $("#add").click(function  () {
                $("#table tr:last").before('<tr><td>美的冰箱</td><td>2</td><td>1000</td><td><input type="button"  value="删除" class="delBtn"  /></td></tr>');
            });
            //删除订单:
            //动态生成的按钮,对$().click()是不敏感的,无效的
            //注意：使用 on() 方法添加的事件处理程序适用于当前及未来的元素（比如由脚本创建的新元素）。
            //可以使用on() $(document).on("事件","操作对象",function(){})
            /*$(".delBtn").click(function(){
                console.log("btn click");
            })*/
            $(document).on("click",".delBtn",function(){
                //console.log("btn click");
                //删除这个行
                //$(this)->按钮   按钮.parent()得到父节点->td.parent()->父节点tr
                $(this).parent().parent().remove();
            })
        })
    </script>
